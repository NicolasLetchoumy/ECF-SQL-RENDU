customers = cette table regroupe tout les clients elle est en lien avec :
- orders via customerNumber
- payments via customerNumber

employees = cette table regroupe tout les employer de l'entreprise elle est en lien avec:
- offices via officeCode


offices = cette table regroupe les informations des entreprises elle est en lien avec :
- employees via officeCode
- customers via PostalCode, phone et city

orderdetail = cette table regroupe tout les commande passer avec les quantité et le numéro de commande elle est en lien avec:
- orders via orderNumber
- products via productCode

orders = cette table regroupe toutes les commandes passer avec les date et qui les a commander elle est en lien avec :
-customers via customerNumber
-orderDetail via orderNumber

payments =  cette table regrouppe les payment efféctuer par les client ainsi que la date elle est en lien avec:
-customers via customerNumber

productlines = cette table regroupe les types de véhicule en vente elle est en lien avec :
-products via productLine

products = cette table regroupe les modèle de véhicule avec leur stock elle est en lien avec:
-orderDetail via productCode
- productlines via productLine
